# -----------------------------------------------------
# Checkout a copy of our source code.
#[user@builder]

    source "${HOME:?}/enteucha.settings"
    if [ ! -e "${ENTEUCHA_CODE:?}" ]
    then
        if [ ! -e "$(dirname ${ENTEUCHA_CODE:?})" ]
        then
            mkdir -p "$(dirname ${ENTEUCHA_CODE:?})"
        fi
        pushd "$(dirname ${ENTEUCHA_CODE:?})"
            git clone "${ENTEUCHA_REPO:?}" "$(basename ${ENTEUCHA_CODE:?})"

        popd
    else
        pushd "${ENTEUCHA_CODE:?}"
            git pull
        popd
    fi
